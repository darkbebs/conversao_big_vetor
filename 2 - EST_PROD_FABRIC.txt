SELECT
     fabricantes_id AS CD_FABRIC
     ,0 AS CD_COMPRADOR
     ,descricao AS NM_FABRIC
     ,coalesce(cgc, '00000000000000') AS CGC_FABRIC
     ,0 AS FREQ_REV
     ,0 AS LEAD_TIME
     ,0 AS TIPO_FRETE
     ,0 AS COMPRA_EMBALADO
     ,0 AS COMPRA_MULTIPLO
     ,0 AS PERC_DESC_COMERCIAL
     ,0 AS PERC_DESC_FINANCEIRO
     ,0 AS PRZ_PGTO
     ,0 AS SEGUNDA
     ,0 AS TERCA
     ,0 AS QUARTA
     ,0 AS QUINTA
     ,0 AS SEXTA
     ,0 AS SABADO
     ,0 AS DOMINGO
     ,0 AS PARAM_EST
     ,NULL AS DT_AGENDAMENTO_DEFAULT
     ,0 AS INDICADOR_SUPRIMENTO
     ,1 AS CD_USU
     ,now() AS DT_CAD
     ,NULL AS DT_ULT_ALT
     ,0 AS QT_DIAS_ACRES_AGENDA
     ,NULL AS DT_VALID_ACRES_AGENDA
     ,0 AS FLAG_GERAR_AGENDA
     ,0 AS FLAG_EXPORTA_NEOGRID

FROM fabricantes
where apagado = 'N'
